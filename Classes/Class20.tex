\begin{ex}
$\begin{bmatrix}
3 & -1 & 2 & -5\\
0 & 5 & -3 & -6\\
-6 & 7 & -7 &4\\
-5& -8&0 &9\\
\end{bmatrix} = \begin{bmatrix}
3 & -1 & 2 & -5\\
0 & 5 & -3 & -6\\
0 & 5 & -3 &-6\\
-5& -8&0 &9\\
\end{bmatrix} = \begin{bmatrix}
3 & -1 & 2 & -5\\
0 & 5 & -3 & -6\\
0 & 0 & 0 &0\\
-5& -8&0 &9\\
\end{bmatrix} =0$
\end{ex}
